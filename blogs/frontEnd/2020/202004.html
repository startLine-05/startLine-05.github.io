# 6 使用其他的小程序的API

这里需要注意的是，以命令行创建的 ts 项目只会包含 uni-app 本身的 @types 声明，如果你想直接使用 微信小程序（wx） 或其他小程序的 api 的话就需要自行添加，以微信小程序为例：

yarn add @types/weixin-app
1

然后在tsconfig.json中添加该声明后重启 vscode 即可：

"types": [
      "webpack-env",
      // 添加 weixin 的声明
      "weixin-app",
      "@dcloudio/types/uni-app"
]

# 7 使用三方Ui 库

以 uView 库为例子

  • 下载 uView 库到 src 源码根路径 与main.ts 同级
  • 配置 sfc.d 文件 添加文件声明
declare module "*.vue" {
    import Vue from 'vue'
    export default Vue
}
//新增 js
declare module 'uview-ui';

  • 接下来可以像 uView 官网配置 uView 文件 uView 官网
  • main.TS 代码示例
import Vue from 'vue'
import App from './App.vue'

import uView from "uview-ui";
Vue.use(uView);

import store from './store'
Vue.prototype.$store = store

Vue.config.productionTip = false

const app = new App()

// http拦截器，此为需要加入的内容，如果不是写在common目录，请自行修改引入路径
import httpInterceptor from '@/utils/request'
// 这里需要写在最后，是为了等Vue创建对象完成，引入"app"对象(也即页面的"this"实例)
Vue.use(httpInterceptor, app)
//接入全局api
import httpApi from '@/api/api'
Vue.use(httpApi, app)

app.$mount()
